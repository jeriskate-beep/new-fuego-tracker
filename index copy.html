        const App = () => {

            const [currentWeek, setCurrentWeek] = useState(0);

            const [progress, setProgress] = useState(() => {

                // Load progress from localStorage or initialize empty

                const saved = localStorage.getItem('fuegoTrainingProgress');

                return saved ? JSON.parse(saved) : trainingPlan.map(week => week.map(() => ({ completed: [], notes: '' })));

            });


            // Save progress to localStorage whenever it changes

            useEffect(() => {

                localStorage.setItem('fuegoTrainingProgress', JSON.stringify(progress));

            }, [progress]);


            const handleComplete = (weekIndex, dayIndex, workoutIndex) => {

                const newProgress = [...progress];

                const dayProgress = { ...newProgress[weekIndex][dayIndex] };

                if (!dayProgress.completed.includes(workoutIndex)) {

                    dayProgress.completed.push(workoutIndex);

                } else {

                    dayProgress.completed = dayProgress.completed.filter(i => i !== workoutIndex);

                }

                newProgress[weekIndex][dayIndex] = dayProgress;

                setProgress(newProgress);

            };


            const handleNotesChange = (weekIndex, dayIndex, notes) => {

                const newProgress = [...progress];

                newProgress[weekIndex][dayIndex] = { ...newProgress[weekIndex][dayIndex], notes };

                setProgress(newProgress);

            };


            return (

                <div className="max-w-4xl mx-auto">

                    <h1 className="text-2xl font-bold text-center mb-4">Fuego Hike Training Tracker</h1>

                    <div className="flex justify-between mb-4">

                        {trainingPlan.map((week, index) => (

                            <button

                                key={index}

                                className={`px-2 py-1 rounded ${currentWeek === index ? 'bg-blue-500 text-white' : 'bg-gray-200'}`}

                                onClick={() => setCurrentWeek(index)}

                            >

                                Week {index + 1}

                            </button>

                        ))}

                    </div>

                    <div className="space-y-4">

                        {trainingPlan[currentWeek].map((day, dayIndex) => (

                            <div key={day.day} className="bg-white p-4 rounded shadow">

                                <h2 className="text-xl font-semibold mb-2">{day.day}</h2>

                                <h3 className="font-medium">Workouts & Ankle Exercises:</h3>

                                <ul className="list-disc pl-5 mb-2">

                                    {day.workouts.map((workout, workoutIndex) => (

                                        <li key={workoutIndex} className="flex items-center">

                                            <input

                                                type="checkbox"

                                                checked={progress[currentWeek][dayIndex].completed.includes(workoutIndex)}

                                                onChange={() => handleComplete(currentWeek, dayIndex, workoutIndex)}

                                                className="mr-2"

                                            />

                                            <span>{workout}</span>

                                        </li>

                                    ))}

                                </ul>

                                <h3 className="font-medium">Nutrition:</h3>

                                <p className="mb-2">{day.nutrition}</p>

                                <h3 className="font-medium">Recovery:</h3>

                                <p className="mb-2">{day.recovery}</p>

                                <h3 className="font-medium">Notes:</h3>

                                <textarea

                                    className="w-full p-2 border rounded"

                                    value={progress[currentWeek][dayIndex].notes}

                                    onChange={(e) => handleNotesChange(currentWeek, dayIndex, e.target.value)}

                                    placeholder="Log pack weight, climbing routes, or how you felt..."

                                />

                            </div>

                        ))}

                    </div>

                </div>

            );

        };
